alter procedure SP_insertarInsumo
(
	@COD_ERP VARCHAR(200),
	@NOMBRE VARCHAR(200),	
	@RPTA int output
)
as
BEGIN
	if exists(select * from INSUMO where COD_ERP=@COD_ERP)
	BEGIN
		UPDATE INSUMO SET NOMBRE=@NOMBRE WHERE COD_ERP=@COD_ERP
		SELECT @RPTA=1
	END
	ELSE
	BEGIN
		INSERT INTO INSUMO 
		(COD_ERP,NOMBRE,FECHA_MODIFICACION)
		VALUES 
		(@COD_ERP,@NOMBRE,GETDATE())
		SELECT @RPTA=1
	END
END
