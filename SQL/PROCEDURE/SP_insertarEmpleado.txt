ALTER procedure SP_insertarEmpleado
(
	@DNI INT,
	@NOMBRE VARCHAR(200),
	@APELLIDO VARCHAR(200),
	@RPTA INT OUTPUT
)
as
BEGIN
	if exists(select * from EMPLEADO where DNI=@DNI)
	BEGIN
		UPDATE EMPLEADO SET APELLIDO=@APELLIDO, NOMBRE=@NOMBRE WHERE DNI=@DNI
		SELECT @RPTA= 1
	END
	ELSE
	BEGIN
		INSERT INTO EMPLEADO 
		(DNI,NOMBRE,APELLIDO,ESTADO,FECHA_MODIFICACION)
		VALUES 
		(@DNI,@NOMBRE,@APELLIDO,1,GETDATE())
		SELECT @RPTA = 1
	END
END
